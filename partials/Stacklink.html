class Node(object):
    def __init__(self,data,next = None):
        self.data = data
        self.next = next

class Stacklink(object):
    def __init__(self):
        self.head = 0
        self.top = -1

    
    def PushStacklink(self,data):
        newnode = Node(data)
        p = self.head
        newnode.next = p
        self.head = newnode
        self.top = self.top + 1

    
    def PopStacklink(self):
        p = self.head
        self.head = p.next
        self.top = self.top - 1

    
    def PrintStacklink(self):
        p = self.head
        temp = self.top
        while self.top != -1:
